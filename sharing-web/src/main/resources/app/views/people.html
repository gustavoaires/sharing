Code example:

// List people in circles
PeoplePage page = google
   .plusOperations()
   .getPeopleInCircles("me", null);

// Search people
PeoplePage page = google
   .plusOperations()
   .searchPeople("john smith", null);

// Get authenticated user profile
Person me = google
   .plusOperations()
   .getGoogleProfile();

// Get activity +1'ers
PeoplePage page = google
   .plusOperations()
   .getActivityPlusOners(
      "z13buzzoqlnoetyuu23ds5w5yz3gwpe3d", null);

// Get activity resharers
PeoplePage page = google
   .plusOperations()
   .getActivityResharers(
